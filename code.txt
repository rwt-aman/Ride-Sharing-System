function formatTimeTo12Hour(time24) {
  const [hours, minutes] = time24.split(':');
  let hour = parseInt(hours);
  const ampm = hour >= 12 ? 'PM' : 'AM';
  hour = hour % 12 || 12;
  return `${hour}:${minutes} ${ampm}`;
}

// Load rider's bookings - UPDATED with Accept/Reject buttons
async function loadRiderBookings() {
  const bookingsList = document.getElementById('bookingsList');
  const bookingsSection = document.getElementById('riderBookings');
  
  bookingsList.innerHTML = '<p>Loading bookings...</p>';
  
  try {
    const response = await fetch(`/rider-bookings?studentId=${currentUser.studentId}`);
    const data = await response.json();
    
    if (data.bookings && data.bookings.length > 0) {
      bookingsSection.classList.remove('hidden');
      
      bookingsList.innerHTML = data.bookings.map(booking => {
        let statusBadge = '';
        let actionButtons = '';
        
        if (booking.status === 'pending') {
          statusBadge = '<span class="status-badge pending">Pending</span>';
          actionButtons = `
            <button class="accept-btn" onclick="acceptBooking(${booking.bookingId})">Accept</button>
            <button class="reject-btn" onclick="rejectBooking(${booking.bookingId})">Reject</button>
          `;
        } else if (booking.status === 'accepted') {
          statusBadge = '<span class="status-badge accepted">Accepted</span>';
        } else if (booking.status === 'rejected') {
          statusBadge = '<span class="status-badge rejected">Rejected</span>';
        }
        
        return `<div class="booking">
          <strong>Seater:</strong> ${booking.seaterName} ${statusBadge}<br>
          <strong>Phone:</strong> <a href="tel:${booking.seaterPhone}">${booking.seaterPhone}</a><br>
          <strong>Destination:</strong> ${booking.destination}<br>
          <strong>Date:</strong> ${booking.rideDate}<br>
          <strong>Time:</strong> ${booking.rideTime}<br>
          <strong>Requested:</strong> ${new Date(booking.bookingTime).toLocaleString()}<br>
          ${actionButtons}
        </div>`;
      }).join('');
    } else {
      bookingsList.innerHTML = '<p>No booking requests yet for your posted rides.</p>';
    }
  } catch (err) {
    bookingsList.innerHTML = '<p>Failed to load bookings.</p>';
  }
}

// Seater search function
async function searchRides() {
  const searchDestination = document.getElementById('searchDestination').value;
  const searchDate = document.getElementById('searchDate').value;
  const resultsDiv = document.getElementById('rideResults');
  
  if (!searchDate) {
    resultsDiv.innerHTML = '<p class="error">Please select a date!</p>';
    return;
  }

  resultsDiv.innerHTML = '<p>Searching...</p>';

  try {
    const response = await fetch(
      `/search-rides?destination=${encodeURIComponent(searchDestination)}&date=${searchDate}`
    );
    const data = await response.json();

    if (Array.isArray(data.rides) && data.rides.length > 0) {
      resultsDiv.innerHTML = data.rides.map((ride, index) =>
        `<div class="ride" id="ride-${index}">
          <strong>Rider:</strong> ${ride.rider_name}<br>
          <strong>Source:</strong> ${ride.source}<br>
          <strong>Destination:</strong> ${ride.destination}<br>
          <strong>Date:</strong> ${ride.ride_date}<br>
          <strong>Time:</strong> ${ride.time_to_leave}<br>
          <strong>Seats Available:</strong> ${ride.seats_available}<br>
          <strong>Phone:</strong> <a href="tel:${ride.phone}">${ride.phone}</a><br>
          <strong>Note:</strong> ${ride.note ? ride.note : 'None'}<br>
          <button class="confirm-btn" onclick="confirmRide(${ride.ride_id}, '${ride.destination}', '${ride.ride_date}', '${ride.time_to_leave}')">
            Request Booking
          </button>
        </div>`
      ).join('');
    } else {
      resultsDiv.innerHTML = '<p>No rides found for this date/destination.</p>';
    }
  } catch (err) {
    resultsDiv.innerHTML = '<p class="error">Failed to search rides!</p>';
  }
}

// Confirm ride booking
function confirmRide(rideId, destination, rideDate, rideTime) {
  if (!currentUser || !currentUser.studentId || !currentUser.fullName || !currentUser.phoneNumber) {
    alert('User information missing. Please logout and login again.');
    return;
  }
  
  if (!confirm('Send booking request for this ride to ' + destination + '?')) {
    return;
  }
  
  const bookingData = {
    rideId: rideId,
    seaterName: currentUser.fullName,
    seaterPhone: currentUser.phoneNumber,
    seaterStudentId: currentUser.studentId,
    destination: destination,
    rideDate: rideDate,
    rideTime: rideTime
  };
  
  fetch('/confirm-booking', {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify(bookingData)
  })
  .then(response => response.json())
  .then(data => {
    if (data.success) {
      alert(data.message || 'Booking request sent! Waiting for rider approval.');
      loadSeaterBookings(); // Refresh seater's bookings
    } else {
      alert('Failed to send request: ' + data.error);
    }
  })
  .catch(err => {
    alert('Failed to contact server!');
  });
}

  try {
    const { studentId, fullName, phoneNumber, email, password } = req.body;

    if (!studentId || !fullName || !phoneNumber || !email || !password) {
      return res.json({ success: false, error: 'All fields are required' });

    const hashedPassword = await bcrypt.hash(password, 10);
    connection = await getConnection();

    await connection.execute(
      `INSERT INTO users (student_id, full_name, phone, email, password)
       VALUES (:studentId, :fullName, :phoneNumber, :email, :password)`,
      { studentId, fullName, phoneNumber, email, password: hashedPassword },
      { autoCommit: true }
    );

    res.json({ success: true, message: 'Registration successful!' });

  } catch (err) {
    if (err.errorNum === 1) {
      res.json({ success: false, error: 'Student ID or Email already exists' });
    } else {
      res.json({ success: false, error: err.message });
    }
  } finally {
    if (connection) await connection.close();
  }
});

        studentId: user.studentId,

    );

    if (bookingResult.rows.length === 0) {
      return res.json({ success: false, error: 'Booking not found' });
    }

    const rideId = bookingResult.rows[0][0];

    // Update booking status to 'accepted'
    await connection.execute(
      `UPDATE bookings SET status = 'accepted' WHERE booking_id = :bookingId`,
      { bookingId },
      { autoCommit: false }
    );

    // Decrease available seats
    await connection.execute(
      `UPDATE rides SET seats_available = seats_available - 1 WHERE ride_id = :rideId`,
      { rideId },
      { autoCommit: false }
    );

    await connection.commit();
    res.json({ success: true, message: 'Booking accepted!' });

  } catch (err) {
    if (connection) await connection.rollback();
    res.json({ success: false, error: err.message });
  } finally {
    if (connection) await connection.close();
  }
});

      return res.json({ success: false, error: 'Booking ID required' });
    }

    connection = await getConnection();

    // Get booking details
    const bookingResult = await connection.execute(
      `SELECT ride_id, status FROM bookings WHERE booking_id = :bookingId`,
      { bookingId }
    );

    if (bookingResult.rows.length === 0) {
      return res.json({ success: false, error: 'Booking not found' });
    }

    const rideId = bookingResult.rows[0][0];
    const status = bookingResult.rows[0][1];

    // Delete the booking
    await connection.execute(
      `DELETE FROM bookings WHERE booking_id = :bookingId`,
      { bookingId },
      { autoCommit: false }
    );

    // If booking was accepted, increase seats back
    if (status === 'accepted') {
      await connection.execute(
        `UPDATE rides SET seats_available = seats_available + 1 WHERE ride_id = :rideId`,
        { rideId },
        { autoCommit: false }
      );
    }

    await connection.commit();
    res.json({ success: true, message: 'Booking cancelled successfully!' });

  } catch (err) {
    if (connection) await connection.rollback();
    res.json({ success: false, error: err.message });
  } finally {
    if (connection) await connection.close();
  }
});

      return res.json({ success: false, error: 'Email and password are required' });
    }

    const result = await pool.query(
      `SELECT student_id, full_name, phone, email, password
       FROM users WHERE email = $1`,
      [email]
    );

    if (result.rows.length === 0) {
      return res.json({ success: false, error: 'Invalid email or password' });
    }

    const user = result.rows[0];
    const isValid = await bcrypt.compare(password, user.password);

    if (!isValid) {
      return res.json({ success: false, error: 'Invalid email or password' });
    }

    res.json({
      success: true,
      message: 'Login successful!',
      user: {
        studentId: user.student_id,
        fullName: user.full_name,
        phoneNumber: user.phone,
        email: user.email
      },
    });

app.post('/accept-booking', async (req, res) => {
  const client = await pool.connect();
  try {
    const { bookingId } = req.body;

    if (!bookingId) {
      return res.json({ success: false, error: 'Booking ID required' });
    }

    await client.query('BEGIN');

    const booking = await client.query(
      `SELECT ride_id FROM bookings WHERE booking_id = $1`,
      [bookingId]
    );

    if (booking.rows.length === 0) {
      await client.query('ROLLBACK');
      return res.json({ success: false, error: 'Booking not found' });
    }

    const rideId = booking.rows[0].ride_id;

    await client.query(
      `UPDATE bookings SET status = 'accepted' WHERE booking_id = $1`,
      [bookingId]
    );

    await client.query(
      `UPDATE rides SET seats_available = seats_available - 1 WHERE ride_id = $1`,
      [rideId]
    );

    await client.query('COMMIT');

    res.json({ success: true, message: 'Booking accepted!' });
  } catch (err) {
    await client.query('ROLLBACK');
    res.json({ success: false, error: err.message });
  } finally {
    client.release();
  }
});

app.post('/cancel-booking', async (req, res) => {
  const client = await pool.connect();
  try {
    const { bookingId } = req.body;

    if (!bookingId) {
      return res.json({ success: false, error: 'Booking ID required' });
    }

    await client.query('BEGIN');

    const booking = await client.query(
      `SELECT ride_id, status FROM bookings WHERE booking_id = $1`,
      [bookingId]
    );

    if (booking.rows.length === 0) {
      await client.query('ROLLBACK');
      return res.json({ success: false, error: 'Booking not found' });
    }

    const { ride_id: rideId, status } = booking.rows[0];

    await client.query(
      `DELETE FROM bookings WHERE booking_id = $1`,
      [bookingId]
    );

    if (status === 'accepted') {
      await client.query(
        `UPDATE rides SET seats_available = seats_available + 1 WHERE ride_id = $1`,
        [rideId]
      );
    }

    await client.query('COMMIT');

    res.json({ success: true, message: 'Booking cancelled successfully!' });
  } catch (err) {
    await client.query('ROLLBACK');
    res.json({ success: false, error: err.message });
  } finally {
    client.release();
  }
});
